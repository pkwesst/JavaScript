package Controller; import java.io.IOException; import java.io.PrintWriter; import javax.servlet.ServletException; import javax.servlet.annotation.WebServlet; import javax.servlet.http.HttpServlet; import javax.servlet.http.HttpServletRequest; import javax.servlet.http.HttpServletResponse; @WebServlet("/ex01sum") public class ex01sum extends HttpServlet { private static final long serialVersionUID = 1L; protected void service(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException { // 1. 파라미터 수집 : 받아온 데이터들을 변수에 넣어서 저장 int num1 = Integer.parseInt(request.getParameter("num1")); //받아온 값은 String 타입이므로 정수형으로 변환 int num2 = Integer.parseInt(request.getParameter("num2")); // 13과 15의 합은 28입니다. String result = num1 + " 과" + num2 + "의 합은" + (num1 + num2) + "입니다."; //괄호 필요 // Ajax 통신에서도 값을 돌려줄 때, PrintWriter 객체 사용 // 2. 인코딩 : Ajax는 utf-8만 지원 response.setCharacterEncoding("utf-8"); // PrintWriter 객체 생성 PrintWriter out = response.getWriter(); // 결과값 전송 out.print(result); } }
